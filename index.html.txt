    <script>
        // Calcular precio total
        document.getElementById('ticketQuantity').addEventListener('input', function() {
            const quantity = parseInt(this.value) || 0;
            const total = quantity * 5;
            document.getElementById('totalPrice').textContent = `Total: S/ ${total.toFixed(2)}`;
        });

        // Abrir modal
        function openModal() {
            document.getElementById('buyModal').style.display = 'flex';
        }

        // Cerrar modal
        function closeModal() {
            document.getElementById('buyModal').style.display = 'none';
        }

        // Confirmar compra
        function confirmPurchase() {
            const quantity = parseInt(document.getElementById('ticketQuantity').value);
            if (quantity > 0 && quantity <= 500) {
                alert(`¡Gracias por tu interés!\n\nHas seleccionado ${quantity} ticket(s)\nTotal a pagar: S/ ${(quantity * 5).toFixed(2)}\n\nPor favor, realiza el pago con Yape al número 904090716 y envía tu comprobante.`);
                closeModal();
                openYape();
            } else {
                alert('Por favor, ingresa una cantidad válida (1-500 tickets)');
            }
        }

        // Abrir Yape
        function openYape() {
            // Intentar abrir la app de Yape
            const yapeNumber = '904090716';
            
            // Para móviles Android
            window.location.href = `yape://send?phone=${yapeNumber}`;
            
            // Fallback: abrir WhatsApp para enviar comprobante
            setTimeout(() => {
                const whatsappMessage = `Hola, deseo participar en el sorteo. Mi número de Yape es: ${yapeNumber}`;
                const whatsappUrl = `https://wa.me/51${yapeNumber}?text=${encodeURIComponent(whatsappMessage)}`;
                
                if (confirm('¿No se abrió Yape? Haz clic en Aceptar para contactar por WhatsApp y enviar tu comprobante.')) {
                    window.open(whatsappUrl, '_blank');
                }
            }, 1000);
        }

        // Descargar Excel
        function downloadExcel() {
            // Crear datos de ejemplo (en un caso real, esto vendría de una base de datos)
            let csvContent = "data:text/csv;charset=utf-8,";
            csvContent += "Nro,Ticket,Nombre,Apellido,Telefono,Email,Fecha de Compra\n";
            
            // Generar 500 participantes de ejemplo
            for (let i = 1; i <= 500; i++) {
                csvContent += `${i},TICKET-${String(i).padStart(4, '0')},Participante ${i},Apellido ${i},999999${String(i).padStart(3, '0')},participante${i}@email.com,04/04/2026\n`;
            }
            
            const encodedUri = encodeURI(csvContent);
            const link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", "lista_participantes_sorteo.csv");
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            
            alert('Se ha descargado el archivo Excel con la lista de participantes');
        }

        // Cerrar modal al hacer clic fuera
        window.onclick = function(event) {
            const modal = document.getElementById('buyModal');
            if (event.target == modal) {
                closeModal();
            }
        }

        // Animación de entrada suave
        window.addEventListener('load', () => {
            document.body.style.opacity = '0';
            setTimeout(() => {
                document.body.style.transition = 'opacity 0.5s ease';
                document.body.style.opacity = '1';
            }, 100);
        });
    </script>
